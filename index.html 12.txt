  cart.push(itemName);
  updateCartDisplay();
}

function updateCartDisplay() {
  cartList.innerHTML = '';
  cart.forEach((item, index) => {
    const li = document.createElement('li');
    li.textContent = `${item}`;
    cartList.appendChild(li);
  });
}
